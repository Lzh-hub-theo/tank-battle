mask_enemy = masks["enemy"]

for cnt in contours:

    center = (x + w//2, y + h//2)

    state['enemy_positions'].append(center)
    state['enemy_count'] += 1

    cv2.rectangle(img, (x, y), (x+w, y+h), (0, 0, 255), 2)
